[docs]def linear_increase(start_retry, max_retry, step,
                    previous_retry, active_count, failed_count, started_count, fallback_count):
    if started_count > 0 or fallback_count > 0:
        retry = previous_retry + step
        if retry > max_retry:
            return max_retry
        else:
            return retry
    else:
        return start_retry




[docs]def linear_adopt(start_retry, max_retry, step,
                 previous_retry, active_count, failed_count, started_count, fallback_count):
    if started_count > 0 or fallback_count > 0:
        retry = previous_retry + step
        if retry > max_retry:
            return max_retry
        else:
            return retry
    else:
        retry = previous_retry - step
        if retry < start_retry:
            return start_retry
        else:
            return retry
