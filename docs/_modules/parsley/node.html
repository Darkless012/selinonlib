[docs]    def check_name(name):
        """
        Check whether name is a correct node (flow/task) name

        :param name: node name
        :return: True if name is a correct node name
        :rtype: bool
        """
        r = re.compile(r"^[_a-zA-Z][_a-zA-Z0-9]*$")
        return r.match(name)
